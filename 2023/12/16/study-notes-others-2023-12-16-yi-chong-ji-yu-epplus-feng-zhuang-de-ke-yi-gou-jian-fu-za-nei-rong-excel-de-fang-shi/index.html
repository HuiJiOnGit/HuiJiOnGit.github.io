一种基于EPPlus封装的可以构建复杂内容Excel的方式

excel Epplus
Epplus
发布日期:   2023-12-16

背景

客户需求一个复杂内容的 Excel 的导出, 涉及到表头复杂, 多行多列, 存在多个行合并,列合并,导出图片等操作

思路

Epplus支持通过索引和 Excel 的位置也就是A1,B2这种方式来选择一个ExcelRange, 通过这种它提供多样的功能,基于此封装一个可以构建复杂表格的方案

代码

public class ExcelCell
{
    /*
    提供多种构造函数,支持根据索引和地址的方式
    */
    public ExcelCell(string adress, string value)
    {
        Adress = adress;
        Value = value;
    }
    public ExcelCell(int rowIndex, int colIndex, string value)
    {
        Adress = GetCellName(rowIndex, colIndex);
        Value = value;
    }
    /// <summary>
    /// 单元格地址A1,B2等
    /// </summary>
    public string Adress { get; set; }
    /// <summary>
    /// 单元格值
    /// </summary>
    public string Value { get; set; }
    /// <summary>
    /// 单元格合并区域,示例A1:B2，为空表示不合并
    /// </summary>
    public string MergeArea { get; set; }
    public MergeMode MergeMode { get; private set; }
    public ExcelHorizontalAlignment HorizontalAlignment { get; set; } = ExcelHorizontalAlignment.Center;
    public ExcelVerticalAlignment VerticalAlignment { get; set; } = ExcelVerticalAlignment.Center;
    public System.Drawing.Color BgColor { get; set; } = System.Drawing.Color.FromArgb(166, 166, 166);
    public System.Drawing.Color FontColor { get; set; } = System.Drawing.Color.Black;
    public bool IsBold { get; set; }
    public ExcelCell RowMerge(int merge)
    {
        MergeArea = GetMergedCellRange(merge, Adress, true);
        MergeMode = MergeMode.Row;
        return this;
    }
    public ExcelCell ColMerge(int merge)
    {
        MergeArea = GetMergedCellRange(merge, Adress, false);
        MergeMode = MergeMode.Col;
        return this;
    }
    public ExcelCell CrossMerge(int rowMerge, int colMerge)
    {
        MergeArea = GetMergedCellRange(rowMerge, colMerge, Adress);
        MergeMode = MergeMode.Cross;
        return this;
    }
    public ExcelCell SetBgColor(string color)
    {
        BgColor = System.Drawing.ColorTranslator.FromHtml(color);
        return this;
    }
    public ExcelCell SetBgColor(System.Drawing.Color color)
    {
        BgColor = color;
        return this;
    }
    public ExcelCell SetFontColor(string color)
    {
        FontColor = System.Drawing.ColorTranslator.FromHtml(color);
        return this;
    }
    public ExcelCell SetFontColor(System.Drawing.Color color)
    {
        FontColor = color;
        return this;
    }
    public ExcelCell SetFontBold(bool bold = true)
    {
        IsBold = bold;
        return this;
    }
    public ExcelCell SetHorizontalAlignment(ExcelHorizontalAlignment horizontalAlignment)
    {
        HorizontalAlignment = horizontalAlignment;
        return this;
    }
    public ExcelCell SetVerticalAlignment(ExcelVerticalAlignment verticalAlignment)
    {
        VerticalAlignment = verticalAlignment;
        return this;
    }
    // 精髓零: 根据行列索引获取excel的位置,感谢ChatGPT
    internal static string GetCellName(int rowIndex, int colIndex)
    {
        string cellName = "";
        while (colIndex > 0)
        {
            int remainder = (colIndex - 1) % 26;
            cellName = (char)('A' + remainder) + cellName;
            colIndex = (colIndex - 1) / 26;
        }
        return cellName + rowIndex.ToString();
    }
    // 精髓一: 根据excel地址转换成行索引和列索引,感谢ChatGPT
    internal static void GetRowIndexAndColIndex(string cellName, out int rowIndex, out int colIndex)
    {
        if (Regex.IsMatch(cellName, @"[A-Z]+\d+"))
        {
            string colPart = Regex.Match(cellName, @"[A-Z]+").Value;
            string rowPart = Regex.Match(cellName, @"\d+").Value;
            colIndex = 0;

            foreach (char c in colPart)
            {
                colIndex = colIndex * 26 + (c - 'A' + 1);
            }

            rowIndex = int.Parse(rowPart);
        }
        else
        {
            throw new ArgumentException("Invalid cell name format.");
        }
    }
    // 精髓二: 获取合并后的范围,感谢ChatGPT
    internal static string GetMergedCellRange(int count, string startCellName, bool mergeMode)
    {
        if (count <= 0)
        {
            throw new ArgumentException("Count must be greater than 0.");
        }
        GetRowIndexAndColIndex(startCellName, out int rowIndex, out int colIndex);
        if (mergeMode == true)
        {
            int endRowIndex = rowIndex + count - 1;
            string endCellName = GetCellName(endRowIndex, colIndex);
            return startCellName + ":" + endCellName;
        }
        else
        {
            int endColIndex = colIndex + count - 1;
            string endCellName = GetCellName(rowIndex, endColIndex);
            return startCellName + ":" + endCellName;
        }
    }
    // 精髓二: 获取合并后的范围,感谢ChatGPT
    static string GetMergedCellRange(int rowCount, int colCount, string startCellName)
    {
        if (rowCount <= 0 || colCount <= 0)
        {
            throw new ArgumentException("Row count and column count must be greater than 0.");
        }
        GetRowIndexAndColIndex(startCellName, out int rowIndex, out int colIndex);
        int endRowIndex = rowIndex + rowCount - 1;
        int endColIndex = colIndex + colCount - 1;
        string endCellName = GetCellName(endRowIndex, endColIndex);
        return startCellName + ":" + endCellName;
    }
}
public enum MergeMode
{
    None = 0,
    Row = 1,
    Col = 2,
    Cross = Row | Col,
}

public static byte[] ExportExcelSingleSheet(DataTable dataSource, ExcelHeaderArea headerArea, string sheetName)
{
    using ExcelPackage package = new();
    // add sheet name
    ExcelWorksheet worksheet = package.Workbook.Worksheets.Add($"{sheetName}Data");
    DrawTitle();
    DrawHeader();
    worksheet.Cells[headerArea.RowCount + 2, 1].LoadFromDataTable(dataSource, false);
    //设置自动列宽
    for (int i = 1; i <= dataSource.Columns.Count; i++)
    {
        worksheet.Column(i).AutoFit();
        worksheet.Column(i).Style.WrapText = true;
    }
    DrawBorder();
    return package.GetAsByteArray();

    void DrawBorder()
    {
        int totalRowQuantity = 1 + headerArea.RowCount + dataSource.Rows.Count;
        var modelTable = worksheet.Cells[1, 1, totalRowQuantity, dataSource.Columns.Count];
        //modelTable.Style.Border.BorderAround(ExcelBorderStyle.Thin, System.Drawing.Color.Black);
        modelTable.Style.Border.Top.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Bottom.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Left.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Right.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Top.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Bottom.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Left.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Right.Color.SetColor(System.Drawing.Color.Black);
    }

    void DrawTitle()
    {
        ExcelRange excelRange = worksheet.Cells[1,1,1, dataSource.Columns.Count];
        excelRange.Merge = true;
        excelRange.Style.Font.Size = 20;
        excelRange.Style.VerticalAlignment = ExcelVerticalAlignment.Center;
        excelRange.Style.HorizontalAlignment = ExcelHorizontalAlignment.Center;
        excelRange.Style.WrapText = false;
        excelRange.Style.ShrinkToFit = true;
        excelRange.Value = sheetName;
        worksheet.Row(1).Height = 30;
    }

    void DrawHeader()
    {
        for (int i = 0; i < headerArea.Headers.Count; i++)
        {
            var header = headerArea.Headers[i];
            worksheet.Cells[header.Adress].Value = header.Value;
            if (!string.IsNullOrEmpty(header.MergeArea))
            {
                worksheet.Cells[header.MergeArea].Merge = true;
            }
            worksheet.Cells[header.Adress].Style.Font.Color.SetColor(System.Drawing.Color.White);
            worksheet.Cells[header.Adress].Style.Font.Bold = true;
            worksheet.Cells[header.Adress].Style.Fill.PatternType = OfficeOpenXml.Style.ExcelFillStyle.Solid;
            worksheet.Cells[header.Adress].Style.Fill.BackgroundColor.SetColor(System.Drawing.ColorTranslator.FromHtml("#1fb5ad"));
            worksheet.Cells[header.Adress].Style.HorizontalAlignment = header.HorizontalAlignment; //水平居中对齐
            worksheet.Cells[header.Adress].Style.VerticalAlignment = header.VerticalAlignment; //垂直居中对齐
        }
    }
}

最精髓的已经在代码中标出, 比如简单的excel,可以直接定义A1,A2的显示内容,不过一般都是通过索引来操作

然后通过ExportExcelSingleSheet方法,即可导出复杂内容的Excel

对比网上那堆你复制我我复制的Epplus的封装,随便举个例子,有巨大的优势.

  1. 这种封装方式并不依赖Epplus本身, 只要你的Excel导出类库支持Excel地址或索引导出, 也就是重新实现一次ExportExcelSingleSheet方法而已.

    甚至,对比网上一大推使用xxxHelper封装,我们更应该依赖抽象,而不是具体内容, 比如我们定义IExcelExporter接口, 在接口里面定义导出方法

    就可以做到面向抽象,而不是直接面对底层类库

  2. 这个类型提供了基础的构建能力,如果需要更多丰富内容的导出,我推荐使用构建者的模式,基于ExcelCell进行更丰富的组合和封装, 不过这个需要导入图片, 对低层类库有强依赖关系,不利于我们基于抽象来封装, 不过也不是不行, 仅导出文本类型,就完全不需要知道底层是什么, 倒是这里导出了图片, 依赖了Worksheet.Drawings.AddPicture这个API

    但是我们也能对此进行封装, 我可以在ExcelCell中添加图片类型,PS:在这里呼吁,不要使用System.Drawing.Common下的图片定义,在Linux下支持非常垃圾, 可以考虑Stream作为参数, 传入图片内容, 再使用底层的API来画图

public sealed class CompetingExcelBuilder : IDisposable
{
    private const string _sheetName = "竞品对比";
    private readonly int _columnQuantity;
    private int _rowIndex = 2;
    private int _columnIndex = 4;
    private int _colEndIndex;
    private string _tempFirstTitle;
    private readonly double _imagePixelRatio = 6.7924d;
    private readonly double _lastCharRowRatio = 2.0d;
    private readonly int _imageQuality = 50;

    public CompetingExcelBuilder(int columnQuantity, ILogger<CompetingExcelBuilder> logger)
    {
        Package = new ExcelPackage();
        Worksheet = Package.Workbook.Worksheets.Add(Name: $"{_sheetName}Data");
        _columnQuantity = columnQuantity;
        _colEndIndex = _columnIndex + columnQuantity - 1;
        Logger = logger;
        ColumnWidthMap = new(columnQuantity);
        for (int i = _columnIndex; i <= _colEndIndex; i++)
        {
            ColumnWidthMap[i] = 20d;
        }
    }

    protected ExcelPackage Package { get; }

    protected ExcelWorksheet Worksheet { get; }
    protected List<ExcelCell> Labels { get; } = new List<ExcelCell>(512);
    protected Dictionary<string, int> RowLabelIndexMap { get; } = new(512);
    public System.Drawing.Color TitleFontColor { get; set; } = System.Drawing.Color.Black;
    public System.Drawing.Color TitleBgColor { get; set; } = System.Drawing.Color.FromArgb(166, 166, 166);
    public System.Drawing.Color TitleBgColor2 { get; set; } = System.Drawing.Color.FromArgb(217, 217, 217);
    /**
    * DPI   Scale ColumnWidth  RowHeight
    * 96dpi 100%  cw=(pix-5)/8 rh=pix*0.75
    */
    public double DefaultShoeHeight { get; set; } = 100d;//cell_h:3.52cm
    public double DefaultChartHeight { get; set; } = 150d;
    public double DefaultDataWidth { get; set; } = 20d;//cell_w:3.97cm

    protected Dictionary<string, int> ChartKeyRowIndexMap { get; } = new(17);
    protected Dictionary<int, double> ColumnWidthMap { get; }
    protected ILogger<CompetingExcelBuilder> Logger { get; }

    #region 构建内容

    public void AddShoePic()
    {
        //var arr = images.ToArray();
        Labels.Add(new ExcelCell(_rowIndex, 1, "鞋图").CrossMerge(4, 3));
        Worksheet.Row(_rowIndex).Height = DefaultShoeHeight;
        Worksheet.Row(_rowIndex + 1).Height = DefaultShoeHeight;
        Worksheet.Row(_rowIndex + 2).Height = DefaultShoeHeight;
        Worksheet.Row(_rowIndex + 3).Height = DefaultShoeHeight;
        _rowIndex += 4;
    }

    public void AddTitleAndLabel(string title, IEnumerable<string> childrenLabels)
    {
        AddFirstLevelTitle(title);
        foreach (var childrenLabel in childrenLabels)
        {
            Labels.Add(new ExcelCell(_rowIndex, 1, childrenLabel).ColMerge(3).SetFontBold(false));
            RowLabelIndexMap.Add($"{title}.{childrenLabel}", _rowIndex);
            _rowIndex++;
        }
    }

    /// <summary>
    /// 生成左边一列行合并，右边列行合并的label
    /// </summary>
    /// <param name="label"></param>
    /// <param name="childrenLabels"></param>
    public void AddLabel(string label, string[] childrenLabels)
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, label).RowMerge(childrenLabels.Length).SetFontBold(false));
        for (int j = 0; j < childrenLabels.Length; j++)
        {
            Labels.Add(new ExcelCell(_rowIndex + j, 2, childrenLabels[j]).ColMerge(2).SetFontBold(false));
            RowLabelIndexMap.Add($"{_tempFirstTitle}.{label}.{childrenLabels[j]}", _rowIndex + j);
        }
        _rowIndex += childrenLabels.Length;
    }

    /// <summary>
    /// 多个左边合并
    /// </summary>
    /// <param name="labels"></param>
    /// <param name="childrenLabels"></param>
    public void AddLabel(IEnumerable<string> labels, string[] childrenLabels)
    {
        foreach (var label in labels)
        {
            AddLabel(label, childrenLabels);
        }
    }

    public void AddFirstLevelTitle(string title)
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, title)
            .ColMerge(3)
            .SetBgColor(TitleBgColor)
            .SetHorizontalAlignment(OfficeOpenXml.Style.ExcelHorizontalAlignment.Left).SetFontBold());
        Labels.Add(new ExcelCell(_rowIndex, _columnIndex, "").ColMerge(_columnQuantity).SetBgColor(TitleBgColor));
        _tempFirstTitle = title;
        _rowIndex++;
    }

    public void AddSecondLevelTitle(string title)
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, title).ColMerge(3).SetBgColor(TitleBgColor2));
        Labels.Add(new ExcelCell(_rowIndex, _columnIndex, "").ColMerge(_columnQuantity).SetBgColor(TitleBgColor2));
        _rowIndex++;
    }

    public void AddPicRow(string label)
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, label).ColMerge(3));
        Worksheet.Row(_rowIndex).Height = DefaultShoeHeight;
        RowLabelIndexMap.Add($"{_tempFirstTitle}.{label}", _rowIndex);
        _rowIndex++;
    }

    public void AddSummary()
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, "小结").ColMerge(3).SetFontBold(false));
        RowLabelIndexMap.Add($"{_tempFirstTitle}.小结", _rowIndex);
        _rowIndex++;
    }

    public void AddChartRow(string label)
    {
        Labels.Add(new ExcelCell(_rowIndex, 1, label).ColMerge(3).SetFontBold(false));
        Labels.Add(new ExcelCell(_rowIndex, _columnIndex, "").ColMerge(_columnQuantity).SetBgColor("#ffffff"));
        string key = $"{_tempFirstTitle}.{label}";
        RowLabelIndexMap.Add(key, _rowIndex);
        _rowIndex++;
    }

    public void AddRowValue(string rowLabel, IEnumerable<string> values)
    {
        if (!RowLabelIndexMap.TryGetValue(rowLabel, out int rowIndex))
        {
            Logger.LogDebug("{label} No Found In RowLabelIndexMap", rowLabel);
            return;
        }
        int index = _columnIndex;
        foreach (var value in values)
        {
            Worksheet.Cells[rowIndex, index].Value = value;
            Worksheet.Cells[_rowIndex, index].Style.WrapText = true;
            index++;
        }
    }

    public void AddPictures(string key, IEnumerable<string> values, bool setNullValueDefaultPicture = true)
    {
        int rowIndex = key switch
        {
            "PicOuterWaist" => 2,
            "PicInnerWaist" => 3,
            "PicBotton" => 4,
            "PicInsole" => 5,
            "PicKeel" => RowLabelIndexMap.GetOrDefault("底部龙骨解剖(mm).龙骨图", -1),
            _ => RowLabelIndexMap.GetOrDefault(key, -1)
        };
        if (rowIndex == -1) return;
        int columnIndex = _columnIndex;
        string filePath, noImageFilePath = Path.Combine(Directory.GetCurrentDirectory(), "wwwroot", "img", "noimage.jpg");
        foreach (var item in values)
        {
            if (string.IsNullOrEmpty(item) || !File.Exists(item))
            {
                if (setNullValueDefaultPicture)
                {
                    filePath = noImageFilePath;
                }
                else
                {
                    columnIndex++;
                    continue;
                }
            }
            else
            {
                filePath = item;
            }

            using ImageCompressResult compressResult = CompressImage(filePath);
            var image = System.Drawing.Image.FromStream(compressResult.Stream, true, true);
            var picture = Worksheet.Drawings.AddPicture($"{key}_{columnIndex}", image);
            //p_w:20d 0.53cm p_h:100d 2.65cm
            picture.SetSize(compressResult.Width, compressResult.Height);
            picture.SetPosition(rowIndex - 1, 10, columnIndex - 1, 3);
            SetColumMaxWidth(columnIndex, compressResult.Width);
            image.Dispose();
            Logger.LogDebug("{img}:CompressWidth:{width},CompressHeigth:{height}", key, compressResult.Width, compressResult.Height);
            columnIndex++;

        }
    }

    private ImageCompressResult CompressImage(string filePath)
    {
        using ImageMagick.MagickImage magick = new(filePath);
        double sourceRatio = Math.Round((double)magick.BaseWidth / (double)magick.Height, 3);
        double width = DefaultShoeHeight * sourceRatio;
        return ImageMagickCompress(magick, (int)width, (int)DefaultShoeHeight);
    }

    private ImageCompressResult CompressImage(Stream stream)
    {
        using ImageMagick.MagickImage magick = new(stream);
        double sourceRatio = Math.Round((double)magick.Width / (double)magick.Height, 3);
        double width = ColumnWidthMap.Values.Sum();
        double height = width / sourceRatio;
        return ImageMagickCompress(magick, (int)width, (int)height);
    }

    private ImageCompressResult ImageMagickCompress(ImageMagick.MagickImage magick, int width, int height)
    {
        magick.Quality = _imageQuality;
        magick.Resize(width, height);
        MemoryStream memoryStream = new();
        magick.Write(memoryStream);
        return new ImageCompressResult(memoryStream, width, height);
    }

    private void SetColumMaxWidth(int columnIndex, double width)
    {
        if (ColumnWidthMap.TryGetValue(columnIndex, out var maxWidth))
        {
            ColumnWidthMap[columnIndex] = width > maxWidth ? width : maxWidth;
        }
        else
        {
            ColumnWidthMap[columnIndex] = width;
        }
    }

    #endregion 构建内容

    #region 构建excel

    public void Dispose()
    {
        Worksheet.Drawings.Dispose();
        Worksheet.Dispose();
        Package.Dispose();
    }

    private void DrawTitle()
    {
        ExcelRange excelRange = Worksheet.Cells[1, 1, 1, _colEndIndex];
        excelRange.Merge = true;
        excelRange.Style.Font.Size = 20;
        excelRange.Style.VerticalAlignment = ExcelVerticalAlignment.Center;
        excelRange.Style.HorizontalAlignment = ExcelHorizontalAlignment.Center;
        excelRange.Style.WrapText = false;
        excelRange.Style.ShrinkToFit = true;
        excelRange.Value = _sheetName;
        Worksheet.Row(1).Height = 30;
    }

    private void DrawBorder()
    {
        int totalRowQuantity = _rowIndex - 1;
        var modelTable = Worksheet.Cells[1, 1, totalRowQuantity, _colEndIndex];
        modelTable.Style.Border.Top.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Bottom.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Left.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Right.Style = ExcelBorderStyle.Thin;
        modelTable.Style.Border.Top.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Bottom.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Left.Color.SetColor(System.Drawing.Color.Black);
        modelTable.Style.Border.Right.Color.SetColor(System.Drawing.Color.Black);
    }

    public void Build()
    {
        DrawTitle();
        for (int i = 0; i < Labels.Count; i++)
        {
            var header = Labels[i];
            Worksheet.Cells[header.Adress].Value = header.Value;
            if (!string.IsNullOrEmpty(header.MergeArea))
            {
                Worksheet.Cells[header.MergeArea].Merge = true;
            }
            Worksheet.Cells[header.Adress].Style.Font.Color.SetColor(System.Drawing.Color.Black);
            Worksheet.Cells[header.Adress].Style.Font.Bold = header.IsBold;
            Worksheet.Cells[header.Adress].Style.Fill.PatternType = OfficeOpenXml.Style.ExcelFillStyle.Solid;
            Worksheet.Cells[header.Adress].Style.Fill.BackgroundColor.SetColor(header.BgColor);
            Worksheet.Cells[header.Adress].Style.HorizontalAlignment = header.HorizontalAlignment; //水平居中对齐
            Worksheet.Cells[header.Adress].Style.VerticalAlignment = header.VerticalAlignment; //垂直居中对齐
        }
        for (int i = 1; i <= 3; i++)
        {
            Worksheet.Column(i).AutoFit();
            Worksheet.Column(i).Style.WrapText = true;
        }
        for (int i = _columnIndex; i <= _colEndIndex; i++)
        {
            if (ColumnWidthMap.TryGetValue(i, out var maxWidth))
            {
                Worksheet.Column(i).Width = maxWidth / _imagePixelRatio;
            }
            Worksheet.Column(i).Style.WrapText = true;
        }
        DrawBorder();
    }

    public void Save(string fileName)
    {
        Package.SaveAs(new FileInfo(Path.Combine(fileName)));
    }

    public byte[] SaveAsByteArray()
    {
        return Package.GetAsByteArray();
    }

    #endregion 构建excel
}
